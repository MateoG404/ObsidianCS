Exists 3 groups :
1. Verbs end with **-ER** 
2. Verbs end with -IR
3. Verbs end with -RE / -RI / -OIR

## Rule for the verbs with end with -ER #flascard 

Je -> ( -e )
Tu -> ( -es )
Il/Elle -> ( -e )
Nous -> ( -ons )
Vous -> ( -ez ) 
Ils / Elles -> ( -ent )

## Rule for the verbs with end -IR #flascard 

Je -> ( -is )
Tu -> ( -is )
Il/Elle -> ( -it )
Nous -> ( -issons )
Vous -> ( -issez ) 
Ils / Elles -> ( -issent )

## Rule for the verbs with end -RE,-RI, -OIR #flascard 

Je -> ( -s )
Tu -> ( -s )
Il/Elle -> ( - )
Nous -> ( -ons )
